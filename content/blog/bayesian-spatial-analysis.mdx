# Bayesian Spatial Analysis with R-INLA

## Introduction to Bayesian Spatial Analysis

Bayesian spatial analysis allows us to model geographic phenomena considering **uncertainty** and **spatial relationships** between observations.

## Why INLA?

**INLA** (Integrated Nested Laplace Approximations) is a computationally efficient alternative to MCMC for Bayesian models:

- ⚡ **Speed**: 100-1000x faster than MCMC
- 🎯 **Accuracy**: Very precise approximations
- 📈 **Scalability**: Handles thousands of observations
- 🔧 **Ease of use**: Intuitive interface in R

## Practical Example: Air Pollution

Let's analyze PM2.5 pollution data in Spain using a Bayesian spatial model.

### 1. Load and Explore Data

\`\`\`r
# Load spatial data
data <- st_read("pollution_data.shp")

# Basic exploration
summary(data)
head(data)

# Initial visualization
ggplot(data) +
  geom_sf(aes(fill = pm25)) +
  scale_fill_viridis_c(option = "magma") +
  theme_minimal() +
  labs(title = "PM2.5 Levels in Spain",
       fill = "PM2.5 (μg/m³)")
\`\`\`

### 2. Create Spatial Mesh

The mesh is crucial for modeling spatial structure:

\`\`\`r
# Extract coordinates
coords <- st_coordinates(data)

# Create triangulated mesh
mesh <- inla.mesh.2d(
  loc = coords,
  max.edge = c(50, 100),    # Maximum triangle size
  cutoff = 20,               # Minimum distance between points
  offset = c(100, 200)       # Domain extension
)

# Visualize mesh
plot(mesh)
points(coords, col = "red", pch = 16)
\`\`\`

### 3. Define the Model

\`\`\`r
# Create SPDE (Stochastic Partial Differential Equation)
spde <- inla.spde2.matern(mesh = mesh, alpha = 2)

# Indices for spatial field
A <- inla.spde.make.A(mesh = mesh, loc = coords)
w.index <- inla.spde.make.index(name = "spatial.field", 
                                 n.spde = spde$n.spde)

# Data stack
stack <- inla.stack(
  data = list(y = data$pm25),
  A = list(A, 1),
  effects = list(
    c(w.index, list(Intercept = 1)),
    list(
      temperature = data$temperature,
      humidity = data$humidity,
      wind_speed = data$wind_speed
    )
  ),
  tag = "est"
)

# Model formula
formula <- y ~ -1 + Intercept + temperature + humidity + wind_speed +
               f(spatial.field, model = spde)
\`\`\`

### 4. Fit the Model

\`\`\`r
# Fit model
result <- inla(
  formula,
  family = "gaussian",
  data = inla.stack.data(stack),
  control.predictor = list(A = inla.stack.A(stack)),
  control.compute = list(
    dic = TRUE,
    waic = TRUE,
    cpo = TRUE,
    config = TRUE
  ),
  verbose = TRUE
)

# View summary
summary(result)
\`\`\`

## Results Interpretation

### Fixed Effects

\`\`\`r
# Coefficients table

### Spatial Field

\`\`\`r
# Project spatial field onto grid
proj <- inla.mesh.projector(mesh, dims = c(200, 200))
spatial_field <- inla.mesh.project(proj, result$summary.random$spatial.field$mean)

# Visualize
image.plot(spatial_field, col = viridis(100),
           main = "Estimated Spatial Field")

## Prediction at New Locations

\`\`\`r
# New locations for prediction
new_coords <- expand.grid(
  lon = seq(min(coords[,1]), max(coords[,1]), length = 50),
  lat = seq(min(coords[,2]), max(coords[,2]), length = 50)
)

# Projection for predictions
A_pred <- inla.spde.make.A(mesh = mesh, loc = as.matrix(new_coords))

# Prediction stack
stack_pred <- inla.stack(
  data = list(y = NA),
  A = list(A_pred, 1),
  effects = list(
    c(w.index, list(Intercept = 1)),
    list(
      temperature = mean(data$temperature),
      humidity = mean(data$humidity),
      wind_speed = mean(data$wind_speed)
    )
  ),
  tag = "pred"
)

# Combine stacks
stack_full <- inla.stack(stack, stack_pred)

# Refit model
result_pred <- inla(
  formula,
  family = "gaussian",
  data = inla.stack.data(stack_full),
  control.predictor = list(
    A = inla.stack.A(stack_full),
    compute = TRUE,
    link = 1
  )
)

# Extract predictions
index_pred <- inla.stack.index(stack_full, "pred")$data
predictions <- result_pred$summary.fitted.values[index_pred,]
\`\`\`
